cmake_minimum_required(VERSION 3.26)
project(InfixEval)

set(CMAKE_CXX_STANDARD 17)

include_directories(/usr/local/include/)

set(CXXTEST_PYTHON_TESTGEN_EXECUTABLE /usr/local/Cellar/cxxtest/4.4_3/bin/cxxtestgen)

find_package(CxxTest REQUIRED)
# Enable testing
enable_testing()
# Generate the runner.cpp and create the test executable
CXXTEST_ADD_TEST(unittest # Name of the test target
        runner.cpp # This file will be generated by cxxtestgen
        ${CMAKE_CURRENT_SOURCE_DIR}/Infix_Evaluator.h # Source file for your project
        ${CMAKE_CURRENT_SOURCE_DIR}/Infix_Evaluator.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/newCxxTest.h # CxxTest description file (your test suite)
)

add_executable(InfixEval runner.cpp Infix_Evaluator.h Infix_Evaluator.cpp)
